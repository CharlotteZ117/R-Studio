library(ggplot2)

#pie chart
df<-data.frame(
    gender=c("Female","Male"),
    freq=c(87,157),    #no <-, will get wrong answers
    percentage=c(.36,.64)
)
df
##   gender freq percentage
## 1 Female   87       0.36
## 2   Male  157       0.64

###pie chard

ggplot(data=df,aes(x="",y=freq,fill=gender)) +
       geom_col(col="black") +    #change the boundary color
      
       coord_polar(theta="y")

ggplot(data=df,aes(x="", y=freq, fill=gender)) +
      geom_col(col="black") +
       geom_text(aes(label=paste0(percentage*100,"%")),position=position_stack(vjust=0.5))+
      scale_fill_manual(values=c("pink","blue")) +
      coord_polar(theta="y") +
      theme_void() +
      theme(legend.position= "none")
