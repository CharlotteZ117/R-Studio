# Boxplot
ggplot(data=df_m, aes(x="", y=value, fill=variable)) + 
  geom_boxplot() +
  facet_wrap(~variable)+
  
# Change vertical spacing between facets
  theme(panel.spacing = unit(3, "cm"),
        legend.position = "none")

# Q-Q plot
ggplot(data = df_m, aes(sample=value)) +
  geom_qq() + 
  geom_qq_line(color="red") + 
  xlab("Theoretical") +
  ylab("Sample") +
